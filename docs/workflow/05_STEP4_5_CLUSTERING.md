def perform_clustering(
    matrix: pd.DataFrame,
    n_clusters: int,
    random_state: int = 42
) -> Tuple[np.ndarray, np.ndarray]:
    """
    K-Means í´ëŸ¬ìŠ¤í„°ë§ì„ ìˆ˜í–‰í•©ë‹ˆë‹¤.

    """
    # ì •ê·œí™”
    scaler = StandardScaler()
    scaled_data = scaler.fit_transform(matrix)
    
    # K-Means

        n_clusters=n_clusters,
